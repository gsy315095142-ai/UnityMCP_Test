```csharp
    {
        var fullPrompt = PromptBuilder.BuildCodePrompt(prompt, context);

        var requestBody = new
        {
            model = model,
            max_tokens = 4000,
            messages = new[]
            {
                new { role = "user", content = fullPrompt }
```

```csharp
public class AIQuickCommand : EditorWindow
{
    [MenuItem("Tools/AI Generate %#g")] // Ctrl+Shift+G
    public static void ShowWindow()
    {
        var window = GetWindow<AIQuickCommand>();
        window.titleContent = new GUIContent("AI Generate");
        window.ShowPopup();
    }

    private string userInput = "";

    void OnGUI()
    {
        userInput = EditorGUILayout.TextField("Describe:", userInput);

        if (Event.current.type == EventType.KeyDown &&
            Event.current.keyCode == KeyCode.Return)
        {
            GenerateWithAI(userInput);
            Close();
        }
    }
}
```

| **实现** | ComponentConfigurator 扩展 UGUI 组件（Canvas、RectTransform、Button、InputField、Text、Image 等）<br>PrefabGenerator 支持 RectTransform 布局（anchors、sizeDelta）<br>Prompt 中增加 UI 相关说明 |
| **技术要点** | Canvas 需自动添加 EventSystem；Button 需子对象 Text；InputField 结构较复杂 |
| **工作量** | 约 3-5 天 |

### D4. 代码 + 预制体联合生成

| 项目 | 说明 |
|------|------|
| **功能** | 用户输入「创建一个玩家，包含移动脚本和预制体」，AI 先生成 PlayerController.cs，再生成 Player.prefab 并自动挂载该脚本 |
| **实现** | 新增「联合生成」模式；一次请求调用两次 AI（或一次 AI 返回代码+JSON）；生成顺序：先脚本 → 编译等待 → 再预制体并 AddComponent |
| **技术要点** | 需处理 Unity 编译延迟（新脚本生成后需等待编译完成才能 AddComponent） |
| **工作量** | 约 3-4 天 |
